%post -n gpsd
/sbin/chkconfig --add gpsd
/sbin/chkconfig gpsd on
/sbin/ldconfig
if [ -f /etc/udev/rules.d/ ] then echo 'KERNEL="ttyS0", SYMLINK="gps"' >/etc/udev/rules.d/80-gpsd.rules

%postun -n gpsd
/sbin/ldconfig
