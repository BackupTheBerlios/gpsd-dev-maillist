When trying to link on Debian/Woody, there are several missing symbols
during linking.  This patch fixes the problem, by making sure XT_LIBS
are included on the link list.

Index: Makefile.am
===================================================================
--- Makefile.am (revision 1161)
+++ Makefile.am (working copy)
@@ -17,7 +17,7 @@
 #
 gps_SOURCES = display.c display.h gps.c
 gps_CFLAGS = $(X_CFLAGS)
-gps_LDADD = $(LIBM) $(LIBC) $(LIBNSL) $(LIBSOCKET) $(XM_LIBS) $(X_LIBS) $(X_PRE_LIBS) libgps.la -lm
+gps_LDADD = $(LIBM) $(LIBC) $(LIBNSL) $(LIBSOCKET) $(XM_LIBS) $(XT_LIBS) $(X_LIBS) $(X_PRE_LIBS) libgps.la -lm

 #
 # Build xgpsspeed
