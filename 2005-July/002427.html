Since I only have a TSIP-device wich is connected to a tiny embedded
thingy I usually disable all other drivers. TSIP_ENABLE does not cause
BINARY_ENABLE, here's the "fix":

- --- gpsd-2.28~/gpsd.h   2005-07-04 16:34:33.000000000 +0200
+++ gpsd-2.28/gpsd.h    2005-07-11 19:14:55.000000000 +0200
@@ -12,7 +12,7 @@
 #ifdef EARTHMATE_ENABLE
 #define ZODIAC_ENABLE
 #endif
- -#if defined(ZODIAC_ENABLE) || defined(SIRFII_ENABLE) ||
defined(GARMIN_ENABLE)
+#if defined(ZODIAC_ENABLE) || defined(SIRFII_ENABLE) ||
defined(GARMIN_ENABLE) || defined(TSIP_ENABLE)
 #define BINARY_ENABLE
 #endif
 #if defined(TRIPMATE_ENABLE) || defined(BINARY_ENABLE)
