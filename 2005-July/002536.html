__OpenBSD__ should be just what you're looking for in an OpenBSD-only
preprocessor symbol. And below are a couple of ways to generate NaN...

pyxis:ttyp4# ./nan
1 d = 0.000000, f = 0.000000, i = 0
2 d = NaN, f = NaN, i = 0
3 d = 0.000000, f = 0.000000, i = 0
4 d = NaN, f = NaN, i = -2147483648
pyxis:ttyp4# cat nan.c
#include <sys/types.h>
#include <math.h>
#include <stdio.h>

main(){
        double d;
        float f;
        int i;

        d = 0.0; f = 0.0; i = 0; 
        printf ("1 d = %lf, f = %f, i = %d\n", d, f, i); 

        d = (0.0/0.0); f = (0.0/0.0); i = (int)(0.0/0.0); 
        printf ("2 d = %lf, f = %f, i = %d\n", d, f, i); 

        d = 0.0; f = 0.0; i = 0; 
        printf ("3 d = %lf, f = %f, i = %d\n", d, f, i); 

        d = (HUGE_VAL + -HUGE_VAL); f = (HUGE_VAL + -HUGE_VAL); i =
(int)(HUGE_VAL + -HUGE_VAL);
        printf ("4 d = %lf, f = %f, i = %d\n", d, f, i); 
}
