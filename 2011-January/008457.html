And this works as it always has...

perl -MJSON=decode_json -e '
  my $json=decode_json(q{{"obj2":{"123":"val123"},"obj1":{"1":"val1","12":"val12"}}}); 
  print join(",", sort keys %$json), "\n"; 
  print join(",", sort keys %{$json->{obj1}}), "\n"; 
  print $json->{obj1}->{12}, "\n";
'
