Why don't you use the CGI module and use header redirect.

use strict;
use Net::GPSD;
use CGI;

my $query=new CGI;

my $point=$gps->get;

print
$query->redirect("http://maps.google.com/?ll=".$point->lat.",".$point->lon);

'
