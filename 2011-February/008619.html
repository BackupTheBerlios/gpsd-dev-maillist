> size_t strlcat(char *dst, const char *src, size_t siz)
> {
>   size_t slen = strlen(src);
>   size_t dlen = strnlen(dst, siz);
>   if (dlen == siz)
>     --dlen;
>   if (siz != 0) {
>     if (dlen + slen < siz)
>       memcpy(dst + dlen, src, slen + 1);
>     else {
>       memcpy(dst + dlen, src, siz - dlen - 1);
>       dst[siz - 1] = '\0';
>     }
>   }
>   return dlen + slen;
> }
